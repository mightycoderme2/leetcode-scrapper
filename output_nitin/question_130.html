130. Surrounded Regions

Given an m x n matrix board containing 'X' and 'O' , capture all regions that are 4-directionallyÂ surrounded by 'X' .

A region is captured by flipping all 'O' s into 'X' s in that surrounded region.

Example 1: 

Input: board = [["X","X","X","X"],["X","O","O","X"],["X","X","O","X"],["X","O","X","X"]]
Output: [["X","X","X","X"],["X","X","X","X"],["X","X","X","X"],["X","O","X","X"]]
Explanation: Surrounded regions should not be on the border, which means that any 'O' on the border of the board are not flipped to 'X'. Any 'O' that is not on the border and it is not connected to an 'O' on the border will be flipped to 'X'. Two cells are connected if they are adjacent cells connected horizontally or vertically.

Example 2: 

Input: board = [["X"]]
Output: [["X"]]

Constraints:

  • m == board.length
  • n == board[i].length
  • 1 <= m, n <= 200
  • board[i][j] is 'X' or 'O' .

// @lc code=start
class Solution {
public:
    void dfs(vector<vector<char>>& board,int x,int y,char value){
        if(x<0 || y<0 || x>=board.size()|| y>=board[x].size()){
            return;
        }
        if(board[x][y]!='O'){
            return;
        }
        board[x][y] = value;
        dfs(board,x-1,y,value);
        dfs(board,x+1,y,value);
        dfs(board,x,y-1,value);
        dfs(board,x,y+1,value);
    }
    void solve(vector<vector<char>>& board) {
        for(int i = 0; i < board.size();i++){
            for(int j = 0; j < board[i].size();j++){
                if((i==0 || j==0|| i==board.size()-1 || j==board[i].size()-1) && board[i][j]=='O'){
                    dfs(board,i,j,'1');
                }
            }
        }

        for(int i = 1; i < board.size()-1;i++){
            for(int j = 1; j < board[i].size()-1;j++){
                    dfs(board,i,j,'X');
            }
        }

        for(int i = 0; i < board.size();i++){
            for(int j = 0; j < board[i].size();j++){
                if(board[i][j]=='1') board[i][j]='O';
            }
        }
    }
};
