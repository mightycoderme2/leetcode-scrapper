18. 4Sum

Given an array nums of n integers, return an array of all the unique quadruplets [nums[a], nums[b], nums[c], nums[d]] such that:

  • 0 <= a, b, c, dÂ < n
  • a , b , c , and d are distinct .
  • nums[a] + nums[b] + nums[c] + nums[d] == target

You may return the answer in any order .

Example 1: 

Input: nums = [1,0,-1,0,-2,2], target = 0
Output: [[-2,-1,1,2],[-2,0,0,2],[-1,0,0,1]]

Example 2: 

Input: nums = [2,2,2,2,2], target = 8
Output: [[2,2,2,2]]

Constraints:

  • 1 <= nums.length <= 200
  • -10 9 <= nums[i] <= 10 9
  • -10 9 <= target <= 10 9

// @lc code=start
class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int total) {
        sort(nums.begin(), nums.end());
        vector<vector<int>>ans;
        if(nums.size()<4) return ans;
        for(int k=0;k<nums.size();k++){
            int initial = nums[k];
            for(int i = k+1; i < nums.size();i++){
                int target=nums[i];
                int start=i+1;
                int end=nums.size()-1;
                while(start < end ){
                    int sum1=nums[start]+nums[end];
                    int sum2=target+initial;
                    long long sum3=(long long) sum1+ (long long) sum2;
                    if(sum3==total){
                        ans.push_back({target,nums[start],nums[end],initial});
                        int x=nums[start];
                        while(start < end && x==nums[start])
                        start++;
                        x=nums[end];
                        while(start < end && x==nums[end])
                        end--;
                    }
                    else if(sum3>total){
                        end--;
                    }
                    else if(sum3<total){
                        start++;
                    }
                }
                while(i+1<end && nums[i]==nums[i+1]){
                    i++;
                }
            }
                while(k+1<nums.size() && nums[k]==nums[k+1]){
                    k++;
                }
            }
        return ans;
    }
};
