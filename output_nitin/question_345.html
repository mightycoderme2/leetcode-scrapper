345. Reverse Vowels of a String

Given a string s , reverse only all the vowels in the string and return it.

The vowels are 'a' , 'e' , 'i' , 'o' , and 'u' , and they can appear in both cases.

Example 1: 

Input: s = "hello"
Output: "holle"

Example 2: 

Input: s = "leetcode"
Output: "leotcede"

Constraints:

  • 1 <= s.length <= 3 * 10 5
  • s consist of printable ASCII characters. 
/*
 * @lc app=leetcode id=345 lang=cpp
 *
 * [345] Reverse Vowels of a String
 */

// @lc code=start
class Solution {
public:
    string reverseVowels(string s) {
        int start=0,end=s.length()-1;
        while(start<=end){
            while(!(tolower(s[start])=='a'||tolower(s[start])=='e'||tolower(s[start])=='i'||tolower(s[start])=='o'|| tolower(s[start])=='u') && start<=end){
                start++;
            }
            while(!(tolower(s[end])=='a'||tolower(s[end])=='e'||tolower(s[end])=='i'||tolower(s[end])=='o'|| tolower(s[end])=='u') && start<=end){
                end--;
            }
            cout<<start<<end;
            if(start<=end){
            swap(s[start],s[end]);
            start++;end--;
            }
            else{
                return s;
            }
        }
        return s;
    }
};
