504. Base 7

Given an integer num , return a string of its base 7 representation .

Example 1: 

Input: num = 100
Output: "202"

Example 2: 

Input: num = -7
Output: "-10"

Constraints:

  • -10 7 <= num <= 10 7 
/*
 * @lc app=leetcode id=504 lang=cpp
 *
 * [504] Base 7
 */

// @lc code=start
class Solution {
public:
    string convertToBase7(int num) {
        string s="";
        if(num==0) return "0";
        int sign=1;
        if(num<0) sign=-1;
        while(num!=0){
            s=to_string(abs(num%7))+s;
            num=num/7;
        }
        return sign==-1?'-'+s:s;
    }
};
// @lc code=end
