61. Rotate List

Given the head of a linkedÂ list, rotate the list to the right by k places.

Example 1: 

Input: head = [1,2,3,4,5], k = 2
Output: [4,5,1,2,3]

Example 2: 

Input: head = [0,1,2], k = 4
Output: [2,0,1]

Constraints:

  • The number of nodes in the list is in the range [0, 500] .
  • -100 <= Node.val <= 100
  • 0 <= k <= 2 * 10 9

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if(head==NULL)
        {
            return NULL;
        }
        if(head->next==NULL)
        {
            return head;
        }
        ListNode* temp = new ListNode(head->val);
        ListNode *traversenode = head->next;
        int count =1;
        while(traversenode!=NULL )
        {
            ListNode* newnode = new ListNode(traversenode->val);
            newnode->next=temp;
            temp=newnode;
            count++;
            traversenode = traversenode->next;
        }
        cout<<count;
        int main_count = k % count;
        while(main_count)
        {
            ListNode *newnode = new ListNode(temp->val);
            newnode->next=head;
            head=newnode;
            temp=temp->next;
            main_count--;
        }
        temp=head;
        main_count=count;
        while(main_count!=1)
        {
            temp=temp->next;
            main_count--;
        }
        temp->next=NULL;
        return head;
    }
};
