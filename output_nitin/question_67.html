67. Add Binary

Given two binary strings a and b , return their sum as a binary string .

Example 1: 

Input: a = "11", b = "1"
Output: "100"

Example 2: 

Input: a = "1010", b = "1011"
Output: "10101"

Constraints:

  • 1 <= a.length, b.length <= 10 4
  • a and b consistÂ only of '0' or '1' characters.
  • Each string does not contain leading zeros except for the zero itself.

// @lc code=start
class Solution {
public:
    string addBinary(string a, string b) {
        int carry=0;
        string s="";
        if(a.length()>b.length()){
            string str(a.length()-b.length(), '0');
            b=str+b;
        }
        else if(a.length()<b.length()){
            string str(-a.length()+b.length(), '0');
            a=str+a;
        }
        cout<<a<<b;
        for(int i=a.length()-1; i>=0; i--){
                int x=(int)a[i]-48;
                int y=(int)b[i]-48;
                if(x+y+carry==2)
                {
                    s='0'+s;
                    carry=1;
                }
                else if(x+y+carry==3)
                {
                    s='1'+s;
                    carry=1;
                }
                else{
                    s=to_string(x+y+carry)+s;
                    carry=0;
                }
            }
        if(carry==1){
            return '1'+s;
        }
        return s;
    }
};
