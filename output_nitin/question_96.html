96. Unique Binary Search Trees

Given an integer n , return the number of structurally unique BST' s (binary search trees) which has exactly n nodes of unique values from 1 to n .

Example 1: 

Input: n = 3
Output: 5

Example 2: 

Input: n = 1
Output: 1

Constraints:

  • 1 <= n <= 19 
/*
 * @lc app=leetcode id=96 lang=cpp
 *
 * [96] Unique Binary Search Trees
 */

// @lc code=start
class Solution {
public:
    int numTrees(int n) {
        vector<int>dp(n+1,0);
        dp[0]=1;
        dp[1]=1;
        for(int i=2; i<=n; i++){
            for(int j=1; j<=i; j++){
                dp[i]+=dp[i-j]*dp[j-1];
            }
        }
        for(auto y:dp){
                cout<<y<<' ';
        }
        return dp[n];
    }
};
